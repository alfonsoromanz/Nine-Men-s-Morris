394 bool GameModel::isGameOver(int nextFase, int lastPlayer, BoardState state )
395 {
396  // Check if the game is over knowing who was the last player that move
397 
398  if (nextFase == GameController::FASE_SOLUTION)
399  return true;
400 
401  // The total pieces of the opposite player
402  int total;
403  // the source pieces of the opposite player
404  int source;
405  total = state.getTotalByColour( !lastPlayer );
406  source = state.getSourceByColour( !lastPlayer );
407 
408  if ( total < 3)
409  return true;
410  else
411 
412  // If there are pieces in the board
413  if ( total - source > 0 ) {
414  // Check if the opponent has possible movements remaining
415  // In this case the fases are equal for both players
416  if ( nextFase == GameController::FASE_PLACE) {
417  if ( getPossibleFreeMovements(!lastPlayer, state).empty() )
418  return true;

420  if ( getPossibleFlyingMovements(!lastPlayer, state).empty() )
421  return true;
422  } else if ( getPossibleSlidingMovements(!lastPlayer, state).empty() )
